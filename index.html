// ============================================================ // Google Apps Script - ระบบจองรถ นศท. ชั้นปีที่ 2 // โรงเรียนมัธยมวัดหนองแขม ปีการศึกษา 2569 // v2: เพิ่มระบบสถานะอนุมัติการจอง (รออนุมัติ / อนุมัติแล้ว) // ============================================================ // วิธีใช้: // 1. เปิด Google Sheets ที่ใช้งานอยู่ (CAR-RD2) // 2. ไปที่ Extensions > Apps Script // 3. วางโค้ดนี้ทับโค้ดเดิมทั้งหมด // 4. รันฟังก์ชัน migrateAddStatusColumn ครั้งเดียว (เพิ่มคอลัมน์สถานะ // ให้แถวที่จองไว้ก่อนหน้านี้ กลายเป็น "รออนุมัติ" อัตโนมัติ) // 5. กด Deploy > Manage Deployments > แก้ไข (ไอคอนดินสอ) > Version: New // แล้วกด Deploy (URL เดิมจะยังใช้ได้ ไม่ต้องเปลี่ยนใน booking.html) // ============================================================ const SPREADSHEET_ID = SpreadsheetApp.getActiveSpreadsheet().getId(); const MAX_PER_BUS = 23; const TOTAL_BUSES = 7; const ADMIN_PASSWORD = 'SAM2206'; // *** เปลี่ยนรหัสผ่านที่นี่ *** const BUS_GENDER = {1:'ชาย',2:'ชาย',3:'ชาย',4:'ชาย',5:'ชาย',6:'ชาย',7:'หญิง'}; const STATUS_PENDING = 'รออนุมัติ'; const STATUS_APPROVED = 'อนุมัติแล้ว'; // ===== SETUP: สร้าง Sheet อัตโนมัติ ===== function setupSheets() { const ss = SpreadsheetApp.getActiveSpreadsheet(); // Sheet รายการจอง let bookingSheet = ss.getSheetByName('Bookings'); if (!bookingSheet) { bookingSheet = ss.insertSheet('Bookings'); bookingSheet.getRange(1, 1, 1, 8).setValues([[ 'ลำดับ', 'เลข นศท.', 'ชื่อ-นามสกุล', 'ประเภท', 'รถคันที่', 'ที่นั่งที่', 'วันที่จอง', 'สถานะ' ]]); bookingSheet.getRange(1, 1, 1, 8).setFontWeight('bold').setBackground('#1a3a6e').setFontColor('white'); bookingSheet.setFrozenRows(1); } // Sheet สรุปสถานะรถ let statusSheet = ss.getSheetByName('BusStatus'); if (!statusSheet) { statusSheet = ss.insertSheet('BusStatus'); statusSheet.getRange(1, 1, 1, 4).setValues([['รถคันที่', 'ประเภท', 'จองแล้ว', 'ที่นั่งว่าง']]); statusSheet.getRange(1, 1, 1, 4).setFontWeight('bold').setBackground('#1a3a6e').setFontColor('white'); for (let i = 1; i <= TOTAL_BUSES; i++) { statusSheet.getRange(i+1, 1, 1, 4).setValues([[ i, BUS_GENDER[i] || 'ชาย', 0, MAX_PER_BUS ]]); } } return { bookingSheet, statusSheet }; } // ===== MIGRATION: เพิ่มคอลัมน์สถานะให้แถวเก่าที่จองไว้ก่อนหน้านี้ ===== // รันฟังก์ชันนี้ "ครั้งเดียว" จาก Apps Script editor (เลือกฟังก์ชันนี้แล้วกด Run) function migrateAddStatusColumn() { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheetByName('Bookings'); if (!sheet) { setupSheets(); return; } const headerRange = sheet.getRange(1, 1, 1, sheet.getLastColumn()); const headers = headerRange.getValues()[0]; // ถ้ายังไม่มีคอลัมน์ "สถานะ" ให้เพิ่มที่ตำแหน่งคอลัมน์ 8 if (headers.indexOf('สถานะ') === -1) { sheet.getRange(1, 8).setValue('สถานะ'); sheet.getRange(1, 8).setFontWeight('bold').setBackground('#1a3a6e').setFontColor('white'); } const lastRow = sheet.getLastRow(); if (lastRow < 2) return; // ไม่มีข้อมูลผู้จอง // เติมสถานะ "รออนุมัติ" ให้ทุกแถวที่ยังไม่มีค่าสถานะ const statusRange = sheet.getRange(2, 8, lastRow - 1, 1); const values = statusRange.getValues(); const newValues = values.map(row => [row[0] ? row[0] : STATUS_PENDING]); statusRange.setValues(newValues); return { success: true, message: `อัปเดตสถานะให้ ${lastRow - 1} แถวเรียบร้อย` }; } // ===== WEB APP ENTRY POINT ===== function doGet(e) { const action = e.parameter.action; let result; try { switch(action) { case 'status': result = getStatus(); break; case 'book': result = bookSeat(e.parameter); break; case 'check': result = checkBooking(e.parameter); break; case 'passengers':result = getPassengers(e.parameter); break; case 'cancel': result = cancelBooking(e.parameter); break; case 'approve': result = approveBooking(e.parameter); break; case 'unapprove': result = unapproveBooking(e.parameter);break; case 'setup': setupSheets(); result = {success:true, message:'Setup complete'}; break; default: result = { success: false, message: 'Unknown action' }; } } catch(err) { result = { success: false, message: err.toString() }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } // ===== doPost (สำหรับ CORS) ===== function doPost(e) { return doGet(e); } // ===== GET STATUS ===== function getStatus() { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); let sheet = ss.getSheetByName('Bookings'); if (!sheet) { setupSheets(); sheet = ss.getSheetByName('Bookings'); } const data = sheet.getDataRange().getValues(); const buses = {}; for (let i = 1; i <= TOTAL_BUSES; i++) { buses[i] = { count: 0, approvedCount: 0, pendingCount: 0, passengers: [] }; } for (let r = 1; r < data.length; r++) { const row = data[r]; const busNum = parseInt(row[4]); const stdId = row[1]; const name = row[2]; const ts = row[6]; const status = row[7] || STATUS_PENDING; if (busNum >= 1 && busNum <= TOTAL_BUSES) { if (!buses[busNum]) buses[busNum] = { count: 0, approvedCount: 0, pendingCount: 0, passengers: [] }; buses[busNum].count++; if (status === STATUS_APPROVED) buses[busNum].approvedCount++; else buses[busNum].pendingCount++; buses[busNum].passengers.push({ stdId, name, status, timestamp: ts ? Utilities.formatDate(new Date(ts), 'Asia/Bangkok', 'dd/MM/yyyy HH:mm') : '' }); } } return { success: true, buses }; } // ===== BOOK SEAT ===== function bookSeat(params) { const stdId = (params.stdId || '').trim(); const stdName = (params.stdName || '').trim(); const gender = (params.gender || '').trim(); const busNum = parseInt(params.busNum); if (!stdId || !stdName || !gender || !busNum) { return { success: false, message: 'ข้อมูลไม่ครบถ้วน' }; } if (busNum < 1 || busNum > TOTAL_BUSES) { return { success: false, message: 'หมายเลขรถไม่ถูกต้อง' }; } if (gender !== BUS_GENDER[busNum]) { return { success: false, message: `รถคันที่ ${busNum} สำหรับ นศท.${BUS_GENDER[busNum]} เท่านั้น` }; } const ss = SpreadsheetApp.openById(SPREADSHEET_ID); let sheet = ss.getSheetByName('Bookings'); if (!sheet) { setupSheets(); sheet = ss.getSheetByName('Bookings'); } const lock = LockService.getScriptLock(); lock.waitLock(10000); try { const data = sheet.getDataRange().getValues(); for (let r = 1; r < data.length; r++) { if (String(data[r][1]).trim() === stdId) { return { success: false, message: `เลข นศท. ${stdId} ได้จองรถคันที่ ${data[r][4]} ไปแล้ว` }; } } let seatCount = 0; for (let r = 1; r < data.length; r++) { if (parseInt(data[r][4]) === busNum) seatCount++; } if (seatCount >= MAX_PER_BUS) { return { success: false, message: `รถคันที่ ${busNum} เต็มแล้ว (${MAX_PER_BUS}/${MAX_PER_BUS})` }; } const seatNum = seatCount + 1; const newRow = data.length; const timestamp = new Date(); sheet.appendRow([ newRow, stdId, stdName, gender, busNum, seatNum, timestamp, STATUS_PENDING ]); sheet.getRange(sheet.getLastRow(), 7).setNumberFormat('dd/MM/yyyy HH:mm:ss'); return { success: true, seatNum, busNum, status: STATUS_PENDING, message: 'จองสำเร็จ รอการตรวจสอบ/อนุมัติ' }; } finally { lock.releaseLock(); } } // ===== CHECK BOOKING ===== function checkBooking(params) { const stdId = (params.stdId || '').trim(); if (!stdId) return { found: false }; const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheetByName('Bookings'); if (!sheet) return { found: false }; const data = sheet.getDataRange().getValues(); for (let r = 1; r < data.length; r++) { if (String(data[r][1]).trim() === stdId) { return { found: true, stdId: data[r][1], name: data[r][2], busNum: data[r][4], seatNum: data[r][5], status: data[r][7] || STATUS_PENDING, timestamp: data[r][6] ? Utilities.formatDate(new Date(data[r][6]), 'Asia/Bangkok', 'dd/MM/yyyy HH:mm') : '' }; } } return { found: false }; } // ===== GET PASSENGERS (Admin) ===== function getPassengers(params) { const busNum = parseInt(params.busNum); const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheetByName('Bookings'); if (!sheet) return { passengers: [] }; const data = sheet.getDataRange().getValues(); const passengers = []; for (let r = 1; r < data.length; r++) { if (parseInt(data[r][4]) === busNum) { passengers.push({ seatNum: data[r][5], stdId: data[r][1], name: data[r][2], status: data[r][7] || STATUS_PENDING, timestamp: data[r][6] ? Utilities.formatDate(new Date(data[r][6]), 'Asia/Bangkok', 'dd/MM/yyyy HH:mm') : '' }); } } passengers.sort((a, b) => a.seatNum - b.seatNum); return { passengers }; } // ===== APPROVE BOOKING (Admin only - หลังตรวจสอบว่าชำระเงินแล้ว) ===== function approveBooking(params) { if (params.adminPass !== ADMIN_PASSWORD) { return { success: false, message: 'รหัสผ่านผู้ดูแลไม่ถูกต้อง' }; } const stdId = (params.stdId || '').trim(); if (!stdId) return { success: false, message: 'ไม่พบเลข นศท.' }; const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheetByName('Bookings'); if (!sheet) return { success: false, message: 'ไม่พบ Sheet' }; const data = sheet.getDataRange().getValues(); for (let r = 1; r < data.length; r++) { if (String(data[r][1]).trim() === stdId) { sheet.getRange(r + 1, 8).setValue(STATUS_APPROVED); return { success: true, message: `อนุมัติการจองของ ${data[r][2]} แล้ว`, status: STATUS_APPROVED }; } } return { success: false, message: 'ไม่พบเลข นศท. นี้ในระบบ' }; } // ===== UNAPPROVE / ย้อนกลับเป็นรออนุมัติ (กรณีกดอนุมัติผิด) ===== function unapproveBooking(params) { if (params.adminPass !== ADMIN_PASSWORD) { return { success: false, message: 'รหัสผ่านผู้ดูแลไม่ถูกต้อง' }; } const stdId = (params.stdId || '').trim(); if (!stdId) return { success: false, message: 'ไม่พบเลข นศท.' }; const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheetByName('Bookings'); if (!sheet) return { success: false, message: 'ไม่พบ Sheet' }; const data = sheet.getDataRange().getValues(); for (let r = 1; r < data.length; r++) { if (String(data[r][1]).trim() === stdId) { sheet.getRange(r + 1, 8).setValue(STATUS_PENDING); return { success: true, message: `ย้อนสถานะของ ${data[r][2]} เป็นรออนุมัติแล้ว`, status: STATUS_PENDING }; } } return { success: false, message: 'ไม่พบเลข นศท. นี้ในระบบ' }; } // ===== CANCEL BOOKING (Admin) ===== function cancelBooking(params) { if (params.adminPass !== ADMIN_PASSWORD) { return { success: false, message: 'รหัสผ่านผู้ดูแลไม่ถูกต้อง' }; } const stdId = (params.stdId || '').trim(); const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheetByName('Bookings'); if (!sheet) return { success: false, message: 'ไม่พบ Sheet' }; const data = sheet.getDataRange().getValues(); for (let r = 1; r < data.length; r++) { if (String(data[r][1]).trim() === stdId) { sheet.deleteRow(r + 1); return { success: true, message: `ยกเลิกการจองของ ${data[r][2]} สำเร็จ` }; } } return { success: false, message: 'ไม่พบเลข นศท. นี้ในระบบ' }; }
